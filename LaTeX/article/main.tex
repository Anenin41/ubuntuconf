\newcommand{\textBF}[1]{%
    \pdfliteral direct {2 Tr 1 w}               %the second factor is the boldness
     #1%
    \pdfliteral direct {0 Tr 0 w}               %
}

\newcommand{\textDF}[1]{%
    \pdfliteral direct {2 Tr 0.2 w}             %the second factor is the boldness
     #1%
    \pdfliteral direct {0 Tr 0 w}               %
}
